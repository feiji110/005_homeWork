### 定长顺序存储
````c
#include<stdio.h>
#define Status      int
#define OK           1
#define ERROR        0
#define INFEASIBLE  -1
#define MAXSTRLEN       255   
typedef unsigned char SString[MAXSTRLEN + 1];
Status StrAssign(SString T,char * chars)
{ 
	int i,j;
	for(i = 0;chars[i];++i)
	T[0] = 0;
	for(j=0;j<i;j++)
	{
		T[j+1] = chars[j];
		T[0]++;
	}
	return OK;

}
Status StrInsert(SString S, int pos, SString T)
{
	int i,j ; 
	if(pos < 1||pos > S[0])
		return ERROR;
	if(S[0]+T[0] >= MAXSTRLEN )
		return INFEASIBLE;
	for(i = S[0];i != pos-1;--i )
		S[i+T[0]] = S[i];
	for(i = pos,j = 1;j <= T[0];++i,++j )
		S[i] = T[j];
	S[0] += T[0];
	return OK;
}
void PrintStr(SString S)
{
	int i;
	for(i = 1;i <= S[0];++i)
		printf("%c",S[i]);
	printf("\n");
}
int main(void)
{
    SString S, T;
    Status result;
    StrAssign(S, "abcdefhijklmn");
    StrAssign(T, "UVWXYZ");
    result = StrInsert(S, 4, T);
    if(OK == result)
    {
        printf("未发生截断，结果是：");
        PrintStr(S);
    }else if(INFEASIBLE == result){
        printf("发生截断!\n");
     }
    else {
        printf("插入位置有误!\n");
    }

    return 0;
}
````
