//编写一个函数
//求1到任意正整数之间（包含该正整数）的所有质数之和。
function sumPrime(n){
    var sum = 0;
    var isPrime = function(num){
        if(num ==1){
            return false;
        }
        for(var i = 2;i <= Math.sqrt(num);++i){
            if(num % i == 0 ){
                return false;
            }
        }
        return true;
    }
    for(var i = 1;i <= n;++i){
        if(isPrime(i)){
            sum += i;
        }
    }
    return sum;
}
console.log(sumPrime(10));
